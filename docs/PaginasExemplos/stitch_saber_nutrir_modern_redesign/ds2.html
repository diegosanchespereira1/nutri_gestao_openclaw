Como usar

<!-- Card com glow verde -->
<div class="bg-surface-elevated p-6 rounded-3xl border border-border
            glow-card glow-card-green" data-ds-animate>
  ...
</div>

<!-- Card destaque estático (RT, CTA) -->
<div class="bg-gradient-to-br from-primary-dark to-primary-light
            rounded-3xl p-8 text-white glow-card-static">
  ...
</div>

<!-- Botão primário com gradiente -->
<a class="ds-btn ds-btn-primary ds-animate ds-fade-up-sm"
   data-ds-animate href="#">
  Solicitar diagnóstico
</a>

<!-- Hero section -->
<section class="hero-mesh relative min-h-[92vh]">
  <div class="hero-orb hero-orb-1"></div>
  <div class="absolute inset-0 ds-hero-image opacity-30">
    <img src="..." />
  </div>
  <h1>... <span class="text-gradient">highlight</span></h1>
</section>

<!-- Stagger group -->
<div data-ds-animate data-ds-stagger="fade-up">
  <div>Coluna 1</div>
  <div>Coluna 2</div>
</div>
